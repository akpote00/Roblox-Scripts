local function search(text)
	local items = {}
	for i, v in next, path:GetChildren() do
		if text:upper() == v.Name:sub(1, string.len(text)):upper() then
			table.insert(items, v.Name)
		end
	end
	return items
end

	thing.MouseButton1Click:connect(function()
		f()
	end)
end

		spawnframe.Visible = false

selectvehicle.MouseButton1Click:Connect(function()
	if vehiclelistexists == false then
		vehiclelistexists = true
		local VehiclesListFrame = Instance.new("ScrollingFrame", VehiclesPage)
		local function formInstance(class, properties)
			local temp = Instance.new(class)
			for i,v in pairs(properties) do
				temp[i] = v
			end
			return temp
		end

		VehiclesListFrame.Size = UDim2.new(0, 200, 0, 100)
		VehiclesListFrame.BackgroundColor3 = Color3.new(0,0,0)
		VehiclesListFrame.BackgroundTransparency = 0.8
		VehiclesListFrame.Position = UDim2.new(0.225, 0, 0, 25)
		VehiclesListFrame.BottomImage = "rbxasset://textures/ui/Scroll/scroll-middle.png"
		VehiclesListFrame.ScrollBarThickness = 5
		VehiclesListFrame.TopImage = "rbxasset://textures/ui/Scroll/scroll-middle.png"

		local function StartVehiclesList()
			VehiclesListFrame:ClearAllChildren()
			wait()
			local curVehiclesPosition = UDim2.new(0,0,0,0)
			for _,v in pairs(workspace.Vehicles:GetChildren()) do
				local vehicleButton = formInstance('TextButton',
				{BackgroundColor3=Color3.fromRGB(1,1,1),
				BorderSizePixel=0,
				BorderColor3=Color3.fromRGB(1,1,1),
				BackgroundTransparency = 0.8,
				Position=curVehiclesPosition,
				Size=UDim2.new(1,0,0,25),
				ZIndex=2,
				TextSize=14,
				Font=Enum.Font.ArialBold,
				Text=v.Name,
				TextColor3=Color3.new(1,1,1),
				TextTransparency = 0.2,
				TextWrapped=true,
				Parent=VehiclesListFrame})
			curVehiclesPosition = curVehiclesPosition + UDim2.new(0,0,0,25)
			VehiclesListFrame.CanvasSize = UDim2.new(0,0,0,(30*#workspace.Vehicles:GetChildren()))
			vehicleButton.MouseButton1Click:connect(function()
			selectvehicle.Text = v.Name
			VehiclesListFrame:Destroy()
			vehiclelistexists = false
			end)
			end
		end
		StartVehiclesList()
	else
		vehiclelistexists = false
	end
end)
